DROP TABLE IF EXISTS CUSTOMER_RECORDS;

CREATE TABLE CUSTOMER_RECORDS (
    ID INT AUTO_INCREMENT PRIMARY KEY, -- Incrementing ID at the top
    CLIENTS_NAME VARCHAR(255),
    CREDENTIALS VARCHAR(255),
    ADDRESS VARCHAR(255),
    PLAN VARCHAR(100),
    DATE_APPLIED VARCHAR(100),
    CONTACT_NO VARCHAR(50),
    AGE VARCHAR(10),
    SEX VARCHAR(10),
    SOCIAL_MEDIA VARCHAR(255),
    OCCUPATION VARCHAR(255),
    DATE_OF_BIRTH VARCHAR(100),
    PLACE_OF_BIRTH VARCHAR(255),
    OPTICAL_INFO VARCHAR(50),
    SC_CONNECTOR VARCHAR(50),
    FIBER_DROP VARCHAR(50),
    TAPPING_CLIP VARCHAR(50),
    CABLE_TIE VARCHAR(50),
    F_CLAMP VARCHAR(50),
    REMARKS TEXT,
    INSTALLED_BY VARCHAR(255)
);
